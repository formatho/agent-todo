File-Based Persistence The simplest approach: store agent state in local files. Pros: - No external dependencies - Human-readable format - Easy debugging Cons: - File locking issues - No distributed access - Prone to data corruption Implementation: import json import os class FileAgentMemory: def __init__(self, agent_id): self.memory_file = f"/tmp/{agent_id}_memory.json" def save_state(self, state): with open(self.memory_file, 'w') as f: json.dump(state, f, indent=2) def load_state(self): if os.path.exists(self.memory_file): with open(self.memory_file, 'r') as f: return json.load(f) return {} ### 2. Database Persistence More robust: use a proper database for state storage. Pros: - ACID compliance - Concurrent access support - Query capabilities - Data integrity Cons: - Database setup complexity - Connection management - Schema evolution Implementation: import sqlite3 from contextlib import contextmanager class DatabaseAgentMemory: def __init__(self, db_path="agent_memory.db"): self.db_path = db_path self._init_db() def _init_db(self): conn = sqlite3.connect(self.db_path) conn.execute(""" CREATE TABLE IF NOT EXISTS agent_state ( agent_id TEXT, session_id TEXT, key TEXT, value TEXT, timestamp DATETIME DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (agent_id, session_id, key) ) """) conn.close() @contextmanager def _get_connection(self): conn = sqlite3.connect(self.db_path) conn.row_factory = sqlite3.Row try: yield conn finally: conn.close() def save_state(self, agent_id, session_id, state): with self._get_connection() as conn: for key, value in state.items(): conn.execute(""" INSERT OR REPLACE INTO agent_state (agent_id, session_id, key, value) VALUES (?, ?, ?, ?) """, (agent_id, session_id, key, json.dumps(value))) conn.commit() def load_state(self, agent_id, session_id): with self._get_connection() as conn: cursor = conn.execute(""" SELECT key, value FROM agent_state WHERE agent_id = ? AND session_id = ? """, (agent_id, session_id)) return {row['key']: json.loads(row['value']) for row in cursor.fetchall()} ### 3. seconds self._setup_auto_save() def _setup_auto_save(self): def auto_save(): if hasattr(self, 'current_state') and self.current_state: self.save_state() threading.Timer(self.auto_save_interval, auto_save).start() def save_state(self): """Save current working state""" state = { 'timestamp': datetime.now().isoformat(), 'current_task': self.current_task, 'progress': self.progress, 'context': self.context } # Save to persistent storage self._persist_state(state) ### Strategy 2: Checkpoint-Based Persistence Save meaningful milestones. class CheckpointMemoryManager: def __init__(self, agent_id): self.agent_id = agent_id self.checkpoints = [] self.current_checkpoint = None def create_checkpoint(self, name, data): """Create a named checkpoint""" checkpoint = { 'name': name, 'timestamp': datetime.now().isoformat(), 'data': data, 'checksum': self._calculate_checksum(data) } self.checkpoints.append(checkpoint) self._persist_checkpoint(checkpoint) def restore_checkpoint(self, name): """Restore to a named checkpoint""" for checkpoint in reversed(self.checkpoints): if checkpoint['name'] == name: return checkpoint['data'] return None ### Strategy 3: Hybrid Approach Combine file storage with task tracking. class HybridMemoryManager: def __init__(self, agent_id, todo_api_key): self.agent_id = agent_id self.file_memory = FileAgentMemory(agent_id) self.task_memory = AgentTodoMemory(todo_api_key) def save_current_task(self, task_data): """Save current task with all context""" task_data.update({ 'agent_id': self.agent_id, 'memory_file': self.file_memory.memory_file, 'checkpoint': 'current_state' }) return self.task_memory.create_task(task_data) def restore_task(self, task_id): """Restore task with all context""" task = self.task_memory.get_task(task_id) if task: self.file_memory.load_state(task.get('memory_file')) return task ## Best Practices for AI Memory Management ### 1. State Serialization DO: - Use JSON for simple state - Use MessagePack for binary efficiency - Compress large state objects - Version your state schema DON'T: - Store binary blobs in text formats - Use language-specific serialization (Pickle, etc.) - Include sensitive data in memory dumps import msgpack import zlib class EfficientMemorySerializer: @staticmethod def serialize(state): """Serialize state efficiently""" # Convert to MessagePack (binary) packed = msgpack.packb(state) # Compress compressed = zlib.compress(packed) return compressed @staticmethod def deserialize(compressed_data): """Deserialize state efficiently""" # Decompress packed = zlib.decompress(compressed_data) # Unpack MessagePack return msgpack.unpackb(packed) ### 2. Memory Cleanup Automate cleanup to prevent bloat: class MemoryCleanupManager: def __init__(self, max_age_days=30): self.max_age_days = max_age_days def cleanup_old_memories(self): """Remove memories older than max_age_days""" cutoff_date = datetime.now() - timedelta(days=self.max_age_days) old_memories = self._find_memories_before(cutoff_date) for memory in old_memories: if self._is_safe_to_delete(memory): self._delete_memory(memory) def _is_safe_to_delete(self, memory): """Check if memory can be safely deleted""" # Logic to determine if memory is still needed return memory['status'] == 'completed' and memory['age'] > self.max_age_days ### 3. Lazy Loading class LazyMemoryManager: def __init__(self, agent_id): self.agent_id = agent_id self.cached_state = None self.cache_timestamp = None self.cache_ttl = 300 # 5 minutes def get_state(self, key): """Get state value with lazy loading""" if self._should_reload_cache(): self._reload_cache() return self.cached_state.get(key) if self.cached_state else None def _should_reload_cache(self): """Check if cache needs reloading""" return (self.cached_state is None or (datetime.now() - self.cache_timestamp).seconds > self.cache_ttl) ### 2. Memory Compression class CompressedMemoryManager: def __init__(self, compression_level=6): self.compression_level = compression_level def compress_state(self, state): """Compress state before saving""" json_str = json.dumps(state) compressed = zlib.compress(json_str.encode(), level=self.compression_level) return compressed def decompress_state(self, compressed_data): """Decompress state after loading""" json_str = zlib.decompress(compressed_data).decode() return json.loads(json_str) ## Monitoring and Analytics ### Memory Usage Tracking class MemoryMonitor: def __init__(self, agent_id): self.agent_id = agent_id self.metrics = [] def record_memory_usage(self, state_size, operation): """Record memory usage metrics""" metric = { 'timestamp': datetime.now().isoformat(), 'agent_id': self.agent_id, 'state_size_bytes': state_size, 'operation': operation, 'memory_usage_mb': state_size / (1024 * 1024) } self.metrics.append(metric) self._send_to_monitoring(metric) def get_memory_trends(self, hours=24): """Get memory usage trends""" cutoff = datetime.now() -
